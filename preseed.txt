#_preseed_V1

# Set the installation and system language to English.
d-i debian-installer/language string en

# Set the location to United States.
d-i debian-installer/country string US

# Set the keyboard layout to US English.
d-i keyboard-configuration/xkb-keymap select us

# Set the root password to 'password'.
d-i passwd/root-password password password
d-i passwd/root-password-again password password

# Create the initial user account 'andrew' and set the password to 'password'.
d-i passwd/user-fullname string Andrew
d-i passwd/username string andrew
d-i passwd/user-password password password
d-i passwd/user-password-again password password

# Select the internal disk for OS installation.
# Note: This script is probably not robust to all installation contexts; use with care.
# Implements the strategy of selecting the first disk that is not the first USB drive.
d-i partman/early_command string \
    USB_DRIVE_NAME=$(list-devices usb-partition | sed 's/[0-9]//g' | head -n 1); \
    INTERNAL_DISK_NAME=$(list-devices disk | grep -v "$USB_DRIVE_NAME" | head -n 1); \
    debconf-set partman-auto/disk "$INTERNAL_DISK_NAME"

# Use the regular disk partitioning method.
d-i partman-auto/method string regular

# Bypass all precautionary partitioning confirmation prompts.
d-i partman/choose_partition select finish
d-i partman/confirm boolean true

# Bypass final successful installation acknowledgment prompt.
d-i finish-install/reboot_in_progress note

# Power off (and do not restart) the computer once installation has completed.
d-i debian-installer/exit/poweroff boolean true
